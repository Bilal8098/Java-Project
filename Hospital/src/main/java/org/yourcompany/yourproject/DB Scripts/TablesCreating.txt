CREATE SEQUENCE userSeq
START WITH 1
INCREMENT BY 1
NOMAXVALUE
NOCACHE;

CREATE TABLE HospitalUsers(
userId NUMBER DEFAULT userSeq.NEXTVAL PRIMARY KEY,
UserName VARCHAR2(30),
UserAddress VARCHAR2(50),
UserPhone VARCHAR2(20),
UserEMAIL VARCHAR2(30),
UserAge NUMBER,
UserRole VARCHAR2(20)
);

CREATE TABLE Admins(
AdminId NUMBER REFERENCES HospitalUsers(userId),
SALARY NUMBER(10,2)
);

CREATE SEQUENCE DeptSeq
START WITH 1
INCREMENT BY 1
NOMAXVALUE
NOCACHE;

CREATE TABLE Departments(
DeptId NUMBER DEFAULT DEPTSEQ.NEXTVAL PRIMARY KEY,
DeptName VARCHAR(20),
DeptHead NUMBER REFERENCES HospitalUsers(UserId),
NumberOfStaff NUMBER
);

CREATE TABLE DOCTORS(
DocId NUMBER REFERENCES HospitalUsers(userId),
Department NUMBER REFERENCES Departments(DeptId),
Salary NUMBER(10,2)
);

CREATE TABLE Nurses(
nurseId NUMBER REFERENCES HospitalUsers(UserId),
DeptId NUMBER REFERENCES Departments(DeptId),
Shift VARCHAR2(30)
);

CREATE TABLE PATIENT(
PatientId NUMBER REFERENCES HospitalUsers(UserId),
historyDept NUMBER REFERENCES Departments(DeptId)
);

NOCACHE;

INCREMENT BY 1
NOMAXVALUE
NOCACHE;

CREATE TABLE Bills(
BillId NUMBER DEFAULT BILLSEQ.NEXTVAL PRIMARY KEY,

CREATE SEQUENCE RoomSeq
START WITH 1
INCREMENT BY 1
NOMAXVALUE
NOCACHE;

CREATE TABLE ROOMS(
RoomId NUMBER DEFAULT RoomSeq.NEXTVAL PRIMARY KEY,
isOccupied VARCHAR2(5) CHECK(isOccupied = 'Yes' OR isOccupied = 'No') ,
DailyRate NUMBER(10,2),
PatientId NUMBER REFERENCES HospitalUsers(UserId)
