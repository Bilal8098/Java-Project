CREATE OR REPLACE PROCEDURE AddAdmin(
AdminName VARCHAR2,
AdminPass VARCHAR2,
AdminAddress VARCHAR2,
AdminAge NUMBER,
AdminPhone VARCHAR2,
AdminEmail VARCHAR2,
AdminSalary NUMBER
)
IS 
BEGIN
EXECUTE IMMEDIATE 'CREATE USER ' || AdminName || ' IDENTIFIED BY ' || AdminPass;
EXECUTE IMMEDIATE 'ALTER USER ' || AdminName || ' DEFAULT TABLESPACE users QUOTA UNLIMITED ON users';
EXECUTE IMMEDIATE 'ALTER USER ' || AdminName || ' TEMPORARY TABLESPACE TEMP';
EXECUTE IMMEDIATE 'GRANT CONNECT TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT CREATE SESSION, CREATE VIEW, CREATE TABLE, ALTER SESSION, CREATE SEQUENCE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT CREATE SYNONYM, CREATE DATABASE LINK, RESOURCE, UNLIMITED TABLESPACE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT CREATE USER TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT ALTER USER TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT DROP USER TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT GRANT ANY PRIVILEGE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT GRANT ANY ROLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT ALTER ANY ROLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT DROP ANY ROLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT EXECUTE ANY PROCEDURE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT SELECT ANY TABLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT UPDATE ANY TABLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT DELETE ANY TABLE TO ' || AdminName;
EXECUTE IMMEDIATE 'GRANT EXECUTE ANY PROCEDURE TO ' || AdminName;
INSERT INTO HospitalUsers(UserName, UserAddress,
UserAge, UserPhone, UserEmail, UserRole)
VALUES (AdminName, AdminAddress,
AdminAge, AdminPhone, AdminEmail, 'Admin');

INSERT INTO ADMINS(AdminID, SALARY)
VALUES(userseq.currval,AdminSalary);
END;

DECLARE
AdminName VARCHAR2(20);
AdminPass VARCHAR2(20);
AdminAddress VARCHAR2(20);
AdminAge NUMBER ;
AdminPhone VARCHAR2(20);
AdminEmail VARCHAR2(20);
AdminSalary NUMBER;
BEGIN
AddAdmin(
AdminName ,
AdminPass ,
AdminAddress ,
AdminAge ,
AdminPhone ,
AdminEmail ,
AdminSalary 
);

END;
