CREATE OR REPLACE PROCEDURE AddAccountant(
AccId NUMBER,
AccName VARCHAR2,
AccPass VARCHAR2,
AccAddress VARCHAR2,
AccAge NUMBER,
AccPhone VARCHAR2,
AccEmail VARCHAR2,
AccSalary NUMBER
) IS
BEGIN
EXECUTE IMMEDIATE 'CREATE USER ' || AccName || ' IDENTIFIED BY ' || AccPass;
EXECUTE IMMEDIATE 'GRANT CONNECT TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT INSERT ON HOSADMIN.Income TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT INSERT ON HOSADMIN.Expenses TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT INSERT ON HOSADMIN.Bills TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT INSERT ON HOSADMIN.PaymentMethod TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT INSERT ON HOSADMIN.Inventory TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT SELECT ON HOSADMIN.Income TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT SELECT ON HOSADMIN.Expenses TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT SELECT ON HOSADMIN.Bills TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT SELECT ON HOSADMIN.PaymentMethod TO ' || AccName;
EXECUTE IMMEDIATE 'GRANT SELECT ON HOSADMIN.Inventory TO ' || AccName;

INSERT INTO HospitalUsers(UserId, UserName, UserAddress,
UserAge, UserPhone, UserEmail, UserRole)
VALUES (AccId, AccName, AccAddress,
AccAge, AccPhone, AccEmail, 'Accountant');

INSERT INTO Accountants(AccountantId,
Salary) VALUES(AccId, AccSalary);
END;

SHOW ERRORS PROCEDURE AddAccountant;

DECLARE 
ID NUMBER;
NAME VARCHAR2(20);
PASSW VARCHAR2(30);
ADDRESS VARCHAR2(50);
AGE NUMBER;
PHONE VARCHAR2(20);
EMAIL VARCHAR2(30);
SALARY NUMBER(10,2);
BEGIN
AddAccountant(ID, NAME, PASSW, ADDRESS, AGE, PHONE, EMAIL, SALARY);
END;
